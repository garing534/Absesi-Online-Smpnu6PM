function performDelete(id) {

Object.keys(attendanceData).forEach(key => {

if (attendanceData[key].length === 0) {

});

localStorage.setItem('attendanceData', JSON.stringify(attendanceData));

const editModal = document.getElementById('studentModal');

editModal.style.display = 'none';

}

loadStudents();

updateDashboard();

updateStudentDropdown();

// Add to activity log

addRecentActivity(`Siswa ${student.name} (${student.nis}) berhasil dihapus`);

} catch (error) {

console.error('Error deleting student:', error);

alert('Gagal menghapus siswa. Silakan coba lagi.');

}

}

function processImport() {

const fileInput = document.getElementById('importFile');

if (fileInput.files.length > 0) {

// Process file import

const file = fileInput.files[0];

reader.onload = function(e) {

try {

let data = '';

if (file.name.endsWith('.csv')) {

data = e.target.result;

} else {

// For Excel/Word files, we'll treat them as text for this demo

data = e.target.result;

}

processImportData(data);

} catch (error) {

alert('Error membaca file: ' + error.message);

}

};

reader.readAsText(file);

} else if (manualData) {

// Process manual data

processImportData(manualData);

} else {

alert('Pilih file atau masukkan data manual!');

}

}

 

function processImportData(data) {

const lines = data.split('\n').filter(line => line.trim());

let imported = 0;

let errors = [];

 

// Default subjects list

const defaultSubjects = [

'Bahasa Inggris',

'Bahasa Indonesia',

'PKN',

'IPA',

'IPS',

'Seni Budaya',

'PJOK',

'Informatika'

];

 

lines.forEach((line, index) => {

const parts = line.split(',').map(part => part.trim());

if (parts.length >= 4) {

const [nis, name, className, subjectName] = parts;

// Check if NIS already exists

if (students.find(s => s.nis === nis)) {

errors.push(`Baris ${index + 1}: NIS ${nis} sudah terdaftar`);

return;

}

let subjectId, finalSubjectName;

// First try to find in custom subjects

const customSubject = subjects.find(s =>

s.name.toLowerCase() === subjectName.toLowerCase() &&

s.class === className

);

if (customSubject) {

subjectId = customSubject.id;

finalSubjectName = customSubject.name;

} else {

// Check if it's a default subject

const isDefaultSubject = defaultSubjects.some(ds =>

ds.toLowerCase() === subjectName.toLowerCase()

);

if (isDefaultSubject) {

// Use default subject format

const normalizedSubjectName = defaultSubjects.find(ds =>

ds.toLowerCase() === subjectName.toLowerCase()

);

subjectId = `default_${normalizedSubjectName}_${className}`;

finalSubjectName = normalizedSubjectName;

} else {

errors.push(`Baris ${index + 1}: Mata pelajaran "${subjectName}" tidak dikenali`);

return;

}

}

const newStudent = {

id: Date.now() + Math.random(),

nis: nis,

name: name,

class: className,

subjectId: subjectId,

subjectName: finalSubjectName,

createdAt: new Date().toISOString()

};

students.push(newStudent);

imported++;

} else {

errors.push(`Baris ${index + 1}: Format data tidak lengkap`);

}

});

 

localStorage.setItem('students', JSON.stringify(students));

let message = `Import selesai! ${imported} siswa berhasil ditambahkan.`;

if (errors.length > 0) {

message += `\n\nError:\n${errors.join('\n')}`;

}

alert(message);

document.getElementById('importModal').style.display = 'none';

document.getElementById('importFile').value = '';

document.getElementById('manualImportData').value = '';

loadStudents();

updateDashboard();

updateStudentDropdown();

if (imported > 0) {

addRecentActivity(`Import data: ${imported} siswa berhasil ditambahkan`);

}

}

container.innerHTML = '<div class="col-span-full text-center py-8 text-gray-500">Belum ada mata pelajaran. Klik "Tambah Mapel" untuk menambahkan.</div>';

}

 

subjects.forEach(subject => {

const subjectCard = document.createElement('div');

subjectCard.className = 'card p-6';

subjectCard.innerHTML = `

<div class="flex justify-between items-start mb-4">

<div>

<h3 class="font-bold text-lg text-gray-900">${subject.name}</h3>

<p class="text-gray-600">Kelas ${subject.class}</p>

</div>

<button onclick="deleteSubject(${subject.id})" class="text-red-500 hover:text-red-700">

<svg class="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24">

<path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M19 7l-.867 12.142A2 2 0 0116.138 21H7.862a2 2 0 01-1.995-1.858L5 7m5 4v6m4-6v6m1-10V4a1 1 0 00-1-1h-4a1 1 0 00-1 1v3M4 7h16"></path>

</svg>

</button>

</div>

<div class="flex justify-between items-center text-sm text-gray-600">

<span>👥 ${subject.studentCount} siswa</span>

<span>📅 ${new Date(subject.createdAt).toLocaleDateString('id-ID')}</span>

</div>

`;

container.appendChild(subjectCard);

});

}

 

function deleteSubject(id) {

// This could also be converted to a custom modal for consistency

if (confirm('Apakah Anda yakin ingin menghapus mata pelajaran ini?')) {

const subject = subjects.find(s => s.id === id);

subjects = subjects.filter(s => s.id !== id);

localStorage.setItem('subjects', JSON.stringify(subjects));

loadSubjects();

updateSubjectSelects();

updateDashboard();

if (subject) {

addRecentActivity(`Mata pelajaran ${subject.name} berhasil dihapus`);

}

}

}

 

function updateSubjectSelects() {

const selects = ['subjectSelect', 'reportSubject'];

selects.forEach(selectId => {

const select = document.getElementById(selectId);

const currentValue = select.value;

// Clear existing options except the first one

while (select.children.length > 1) {

select.removeChild(select.lastChild);

}

subjects.forEach(subject => {

const option = document.createElement('option');

option.value = subject.id;

option.textContent = `${subject.name} - ${subject.class}`;

select.appendChild(option);

});

select.value = currentValue;

});

}

 

function loadAttendanceData() {

const subjectId = document.getElementById('subjectSelect').value;

const date = document.getElementById('attendanceDate').value;

if (!subjectId || !date) {

document.getElementById('attendanceTableBody').innerHTML =

'<tr><td colspan="7" class="text-center p-8 text-gray-500">Pilih mata pelajaran dan tanggal untuk melihat data absensi</td></tr>';

return;

}

 

const subject = subjects.find(s => s.id == subjectId);

const attendanceKey = `${subjectId}_${date}`;

const attendance = attendanceData[attendanceKey] || [];

 

const tbody = document.getElementById('attendanceTableBody');

tbody.innerHTML = '';

 

// Show actual attendance records

if (attendance.length === 0) {

tbody.innerHTML = '<tr><td colspan="7" class="text-center p-8 text-gray-500">Belum ada data absensi untuk hari ini</td></tr>';

return;

}

 

// Sort attendance by student number

attendance.sort((a, b) => a.studentNumber - b.studentNumber);

 

attendance.forEach(record => {

const row = document.createElement('tr');

row.className = 'border-b hover:bg-gray-50';

const statusBadge = getStatusBadge(record.status);

const methodBadge = record.method === 'manual' ?

'<span class="px-2 py-1 bg-blue-100 text-blue-800 rounded-full text-xs">Manual</span>' :

'<span class="px-2 py-1 bg-green-100 text-green-800 rounded-full text-xs">QR Code</span>';

row.innerHTML = `

<td class="p-3 font-medium">${record.studentNumber}</td>

<td class="p-3">${record.studentName || `Siswa ${record.studentNumber}`}</td>

<td class="p-3">Kelas ${subject.class}</td>

<td class="p-3">${statusBadge}</td>

<td class="p-3">${new Date(record.timestamp).toLocaleString('id-ID')}</td>

<td class="p-3">${methodBadge}</td>

<td class="p-3">

<button onclick="deleteAttendanceRecord(${subjectId}, '${date}', '${record.studentId || record.studentNumber}')" class="text-red-500 hover:text-red-700 text-sm">

Hapus

</button>

</td>

`;

tbody.appendChild(row);

});

}

 

function getStatusBadge(status) {

const badges = {

'hadir': '<span class="px-3 py-1 bg-green-500 text-white rounded-full text-sm font-medium">Hadir</span>',

'sakit': '<span class="px-3 py-1 bg-yellow-500 text-white rounded-full text-sm font-medium">Sakit</span>',

'izin': '<span class="px-3 py-1 bg-blue-500 text-white rounded-full text-sm font-medium">Izin</span>',

'alpha': '<span class="px-3 py-1 bg-red-500 text-white rounded-full text-sm font-medium">Alpha</span>'

};

return badges[status] || '<span class="px-3 py-1 bg-gray-500 text-white rounded-full text-sm font-medium">-</span>';

}

 

function deleteAttendanceRecord(subjectId, date, studentIdentifier) {

if (confirm('Apakah Anda yakin ingin menghapus data absensi ini?')) {

const attendanceKey = `${subjectId}_${date}`;

if (attendanceData[attendanceKey]) {

attendanceData[attendanceKey] = attendanceData[attendanceKey].filter(a =>

a.studentId != studentIdentifier && a.studentNumber != studentIdentifier

);

localStorage.setItem('attendanceData', JSON.stringify(attendanceData));

loadAttendanceData();

updateDashboard();

const subject = subjects.find(s => s.id == subjectId);

addRecentActivity(`Data absensi di ${subject.name} berhasil dihapus`);

}

}

}

 

function startScanner() {

const config = { fps: 10, qrbox: { width: 250, height: 250 } };

html5QrCode = new Html5Qrcode("qr-reader");

html5QrCode.start(

{ facingMode: "environment" },

config,

(decodedText, decodedResult) => {

try {

const qrData = JSON.parse(decodedText);

if (qrData.type === 'student') {

// Process individual student QR

processStudentQRAttendance(qrData);

document.getElementById('scanResultText').textContent =

`Absensi berhasil untuk ${qrData.studentName} - ${qrData.subjectName}`;

} else if (qrData.subjectId && qrData.date) {

// Process class QR

processQRAttendance(qrData);

document.getElementById('scanResultText').textContent =

`Absensi berhasil untuk ${qrData.subjectName} - ${qrData.class}`;

} else {

document.getElementById('scanResultText').textContent = 'QR Code tidak valid untuk absensi';

}

document.getElementById('scanResult').style.display = 'block';

} catch (error) {

document.getElementById('scanResultText').textContent = 'Format QR Code tidak dikenali';

document.getElementById('scanResult').style.display = 'block';

}

},

(errorMessage) => {

// Handle scan error

}

).catch(err => {

console.error('Error starting scanner:', err);

alert('Tidak dapat mengakses kamera. Pastikan izin kamera telah diberikan.');

});

 

document.getElementById('startScanBtn').style.display = 'none';

document.getElementById('stopScanBtn').style.display = 'inline-block';

}

 

function stopScanner() {

if (html5QrCode) {

html5QrCode.stop().then(() => {

html5QrCode.clear();

document.getElementById('startScanBtn').style.display = 'inline-block';

document.getElementById('stopScanBtn').style.display = 'none';

document.getElementById('scanResult').style.display = 'none';

}).catch(err => {

console.error('Error stopping scanner:', err);

});

}

}

 

function processStudentQRAttendance(qrData) {

const today = new Date().toISOString().split('T')[0];

const attendanceKey = `${qrData.subjectId}_${today}`;

if (!attendanceData[attendanceKey]) {

attendanceData[attendanceKey] = [];

}

 

const existingAttendance = attendanceData[attendanceKey].find(a => a.studentId === qrData.studentId);

if (!existingAttendance) {

attendanceData[attendanceKey].push({

studentId: qrData.studentId,

studentNumber: qrData.studentNumber,

studentName: qrData.studentName,

studentNIS: qrData.studentNIS,

status: 'hadir',

timestamp: Date.now(),

method: 'qr'

});

 

localStorage.setItem('attendanceData', JSON.stringify(attendanceData));

updateDashboard();

addRecentActivity(`${qrData.studentName} (NIS: ${qrData.studentNIS}) absen via QR Code individual untuk ${qrData.subjectName}`);

// Show success message

alert(`Absensi berhasil dicatat untuk ${qrData.studentName}!`);

} else {

alert(`${qrData.studentName} sudah melakukan absensi hari ini.`);

}

}

 

function processQRAttendance(qrData) {

// Simulate student scanning (in real app, this would be based on logged-in student)

const studentNumber = Math.floor(Math.random() * 30) + 1; // Random student for demo

const attendanceKey = `${qrData.subjectId}_${qrData.date}`;

if (!attendanceData[attendanceKey]) {

attendanceData[attendanceKey] = [];

}

 

const existingAttendance = attendanceData[attendanceKey].find(a => a.studentNumber === studentNumber);

if (!existingAttendance) {

attendanceData[attendanceKey].push({

studentNumber: studentNumber,

status: 'hadir',

timestamp: Date.now(),

method: 'qr'

});

 

localStorage.setItem('attendanceData', JSON.stringify(attendanceData));

updateDashboard();

addRecentActivity(`Siswa ${studentNumber} absen via QR Code untuk ${qrData.subjectName}`);

}

}

 

function generateReport() {

const subjectId = document.getElementById('reportSubject').value;

const className = document.getElementById('reportClass').value;

const startDate = document.getElementById('reportStartDate').value;

const endDate = document.getElementById('reportEndDate').value;

 

if (!startDate || !endDate) {

alert('Pilih tanggal mulai dan tanggal akhir!');

return;

}

 

const reportContainer = document.getElementById('reportResults');

let reportHTML = '<table class="w-full border-collapse border border-gray-300">';

reportHTML += '<thead><tr class="bg-gray-100">';

reportHTML += '<th class="border border-gray-300 p-3 text-left">Tanggal</th>';

reportHTML += '<th class="border border-gray-300 p-3 text-left">Mata Pelajaran</th>';

reportHTML += '<th class="border border-gray-300 p-3 text-left">Kelas</th>';

reportHTML += '<th class="border border-gray-300 p-3 text-left">Hadir</th>';

reportHTML += '<th class="border border-gray-300 p-3 text-left">Sakit</th>';

reportHTML += '<th class="border border-gray-300 p-3 text-left">Izin</th>';

reportHTML += '<th class="border border-gray-300 p-3 text-left">Alpha</th>';

reportHTML += '<th class="border border-gray-300 p-3 text-left">Total</th>';

reportHTML += '</tr></thead><tbody>';

 

let hasData = false;

 

// Filter subjects based on selection

const filteredSubjects = subjectId ? subjects.filter(s => s.id == subjectId) : subjects;

filteredSubjects.forEach(subject => {

if (className && subject.class !== className) return;

 

// Generate date range

const start = new Date(startDate);

const end = new Date(endDate);

for (let d = new Date(start); d <= end; d.setDate(d.getDate() + 1)) {

const dateStr = d.toISOString().split('T')[0];

const attendanceKey = `${subject.id}_${dateStr}`;

const attendance = attendanceData[attendanceKey] || [];

 

if (attendance.length > 0) {

hasData = true;

const stats = {

hadir: attendance.filter(a => a.status === 'hadir').length,

sakit: attendance.filter(a => a.status === 'sakit').length,

izin: attendance.filter(a => a.status === 'izin').length,

alpha: attendance.filter(a => a.status === 'alpha').length

};

const total = stats.hadir + stats.sakit + stats.izin + stats.alpha;

 

reportHTML += '<tr>';

reportHTML += `<td class="border border-gray-300 p-3">${new Date(dateStr).toLocaleDateString('id-ID')}</td>`;

reportHTML += `<td class="border border-gray-300 p-3">${subject.name}</td>`;

reportHTML += `<td class="border border-gray-300 p-3">${subject.class}</td>`;

reportHTML += `<td class="border border-gray-300 p-3 text-center">${stats.hadir}</td>`;

reportHTML += `<td class="border border-gray-300 p-3 text-center">${stats.sakit}</td>`;

reportHTML += `<td class="border border-gray-300 p-3 text-center">${stats.izin}</td>`;

reportHTML += `<td class="border border-gray-300 p-3 text-center">${stats.alpha}</td>`;

reportHTML += `<td class="border border-gray-300 p-3 text-center font-bold">${total}</td>`;

reportHTML += '</tr>';

}

}

});

 

reportHTML += '</tbody></table>';

 

if (!hasData) {

reportContainer.innerHTML = '<p class="text-gray-500 text-center py-8">Tidak ada data absensi untuk periode yang dipilih</p>';

} else {

reportContainer.innerHTML = reportHTML;

}

}

 

function addRecentActivity(message) {

const container = document.getElementById('recentActivity');

const activityItem = document.createElement('div');

activityItem.className = 'flex items-center gap-3 p-3 bg-gray-50 rounded-lg';

activityItem.innerHTML = `

<div class="w-2 h-2 bg-blue-500 rounded-full"></div>

<div class="flex-1">

<p class="text-sm text-gray-700">${message}</p>

<p class="text-xs text-gray-500">${new Date().toLocaleString('id-ID')}</p>

</div>

`;

if (container.children.length === 1 && container.children[0].textContent.includes('Belum ada aktivitas')) {

container.innerHTML = '';

}

container.insertBefore(activityItem, container.firstChild);

// Keep only last 5 activities

while (container.children.length > 5) {

container.removeChild(container.lastChild);

}

}
